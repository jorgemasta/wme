# Radio Button Group
Radio buttons allow users to select a single answer from a list of multiple.

## Usage
Checkboxes should always be paired with text.

## Example
```
import React from 'react';
import { Radio } from '@mui/material';
import {RadioButtonGroup, FormControlLabel } '@moderntribe/wme';

const MyRadioButtonGroup = () => {
  const [value, setValue] = React.useState('female');

  const handleChange = (event) => {
    setValue(event.target.value);
  };

  return (
    <RadioButtonGroup
      aria-labelledby="demo-radio-buttons-group-label"
      name="radio-buttons-group"
      onChange={handleChange}
      value={value}
      {...args}
    >
      <FormControlLabel key="1" value="Item 1" control={<Radio />} label="female">Female</FormControlLabel>
      <FormControlLabel key="2" value="Item 2" control={<Radio />} label="male">Male</FormControlLabel>
      <FormControlLabel key="3" value="Item 3" control={<Radio />} label="other">Other</FormControlLabel>
    </RadioButtonGroup>
  );
};

export default MyRadioButtonGroup;
```
