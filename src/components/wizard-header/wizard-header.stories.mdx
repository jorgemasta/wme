import {
  ArgsTable,
  Story,
  Canvas,
  Meta
} from '@storybook/addon-docs';
import { useState } from 'react';
import { WizardHeader, ExitButton, Logo } from '..';
import logo from '../../stories/images/logo.png';
import { Box, Typography } from '@mui/material';
import {
  Logout,
  DesktopWindows,
  TabletMac,
  PhoneIphone,
} from '@mui/icons-material';

<Meta
  title="Wizard/WizardHeader"
  component={ WizardHeader }
/>

export const Exit = () => {
  return (
    <Button onClick={() => console.log('Exit Button Clicked')}>
      Exit
      <Logout sx={ { ml: 1 } } />
    </Button>
  );
};

export const DEVICE_SX = {
	active: {
		color: 'primary.dark',
		mx: '5px',
	},
	default: {
		color: '#BFBFBF',
		cursor: 'pointer',
		mx: '5px',
	}
};

export const DEVICES = {
	desktop: DesktopWindows,
	tablet: TabletMac,
	mobile: PhoneIphone,
};

export function ModalDeviceSelection() {
	return (
    <Box sx={ { display: 'flex', alignItems: 'center' }}>
      { Object.keys(DEVICES).map((key) => {
        const DeviceName = DEVICES[ key ];
        return <DeviceName
          key={ key }
          sx={ 'desktop' === key ? DEVICE_SX.active : DEVICE_SX.default }
          onClick={ () => console.log('device changed') }
        />;
      }) }
	  </Box>
  );
}

export const Template = (args) => {
  return (
    <WizardHeader {...args}>
      <Logo logoSrc={logo} logoAlt="Mithrandir Logo" width="150px" />
      <Box>center</Box>
      <ExitButton onClick={() => console.log('Exit Button Clicked')}>
        Exit to Setup
      </ExitButton>
    </WizardHeader>
  );
};

# Wizard Header
The WizardHeader features the logo and the ability to exit. You can pass a component or text as the exit. 

The Wizard component does not require this header, so feel free to build your own if this component doesn't fit your needs.
