# Radio Button Group

Radio buttons allow users to select a single answer from a list of multiple choices.

A styled version of <a href="https://mui.com/material-ui/react-radio-button/" target="_blank">MUI's RadioGroup</a>, so you can use any prop provided there.

## Example

```
import React from 'react';
import { Radio } from '@mui/material';
import { InputLabel, RadioGroup } '@moderntribe/wme';

const MyRadioGroup = () => {
  const [value, setValue] = React.useState('value');

  const handleChange = (event) => {
    setValue(event.target.value);
  };

  return (
    <RadioGroup
      aria-labelledby="demo-radio-buttons-group-label"
      name="radio-buttons-group"
      onChange={handleChange}
      value={value}
      {...args}
    >
      <InputLabel key="1" value="1" control={<Radio />} label="Item One" />
      <InputLabel key="2" value="2" control={<Radio />} label="Item Two" />
      <InputLabel key="3" value="3" control={<Radio />} label="Item Three" />
    </RadioGroup>
  );
};

export default MyRadioGroup;
```
