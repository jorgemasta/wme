export const DEVICE_SX = {
	active: {
		color: 'primary.dark',
		mx: '5px',
	},
	default: {

		cursor: 'pointer',
		mx: '5px',
	}
};

export const DEVICES = {
	desktop: DesktopWindows,
	tablet: TabletMac,
	mobile: PhoneIphone,
};

export function ModalDeviceSelection() {
	return (
    <Box sx={ { display: 'flex', alignItems: 'center' }}>
      { Object.keys(DEVICES).map((key) => {
        const DeviceName = DEVICES[ key ];
        return <DeviceName
          key={ key }
          sx={ 'desktop' === key ? DEVICE_SX.active : DEVICE_SX.default }
          onClick={ () => console.log('device changed') }
        />;
      }) }
	  </Box>
  );
}

export const Template = (args) => {

};

# Wizard Header
The WizardHeader features the logo and the ability to exit. You can pass a component or text as the exit. 

The Wizard component does not require this header, so feel free to build your own if this component doesn't fit your needs.
