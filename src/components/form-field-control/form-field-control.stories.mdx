# Input Label

A styled version of <a href="https://mui.com/base/react-form-control/" target="_blank">MUI's FormControlUnstyled</a>, so you can use any prop provided there.

## Example

```
import React, { useState } from 'react';
import { FormField, FormFieldControl, TextInput } from '@moderntribe/wme';

export const MyFormFieldControl = (props) => (
  <FormFieldControl>
    <FormField
      field={<TextInput />}
      label="Text Field"
    />
  <FormFieldControl>
);

export default MyFormFieldControl;
```
